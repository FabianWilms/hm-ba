% Image Command
% #3 (optional) Size
% #2 IMAGE File
% #3 Caption
\newcommand{\image}[3][]{
	\vspace{1em}
	\begin{minipage}{\linewidth}
		\centering
		
		\ifthenelse{\equal{#1}{}}
		{
			\includegraphics[width=0.9\linewidth]{#2}
		}
		{
			\includegraphics[width=#1\linewidth]{#2}
		}%
		\captionof{figure}{#3}
		\label{fig:#2}
	\end{minipage}
}

% Quote Command
% #1 Quote Text
\newcommand{\myquote}[1]{
	\begin{quote}
		\begin{itshape}
			\enquote{#1}
		\end{itshape}
	\end{quote}
}
